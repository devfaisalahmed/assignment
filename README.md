/*
* Problem 2:  convert degrees to radians.
*/
function degreeTooRadians(degree) {
  const pi = 3.141592653589793;
  let radians = (degree * pi) / 180;
  return radians;
}
console.log("The radians is " + degreeTooRadians(10));
